### 思路
两个链表如果有共同节点，把短的链表和长链表末端对齐，共同节点的位置是相同的。那么可以先计算出两个链表的长度。然让两个指针同时从对齐后相同的位置出发，走相同步会遇到共同节点。
### 解法一：我的
```java
public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        int lengthA = getLength(headA);
        int lengthB = getLength(headB);
        if(lengthA>=lengthB){
            int m = lengthA - lengthB;
            ListNode p = headA;
            ListNode q = headB;
            while(m>0){
                p = p.next;
                m--;
            }
            while(q.next!=null && q!=p){
                p = p.next;
                q = q.next;
            }
            if(p==q)
                return p;
            else
                return null;
        }
        if(lengthA<lengthB){
            int m = lengthB - lengthA;
            ListNode p = headA;
            ListNode q = headB;
            while(m>0){
                q = q.next;
                m--;
            }
            while(q.next!=null && q!=p){
                p = p.next;
                q = q.next;
            }
            if(p==q)
                return p;
            else
                return null;
        }
        return null;
    }

    public int getLength(ListNode list) {
        int length = 0;
        ListNode p = list;
        while (p.next != null) {
            length++;
            p = p.next;
        }
        return length;
    }
}
```
