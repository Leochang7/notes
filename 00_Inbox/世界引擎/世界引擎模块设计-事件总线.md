# 事件总线 (Event Bus)

## 1. 模块概述

事件总线是整个世界引擎的神经系统，负责在全球范围内传递事件、协调各模块之间的通信、处理异步事件流。它是实现分布式仿真、跨区域交互、全局事件传播的关键基础设施。

### 1.1 核心功能

```
事件生产者
    │
    ▼
┌─────────────────────────────────────────────────────────────┐
│                     事件总线                                 │
├─────────────────────────────────────────────────────────────┤
│  1. 事件发布器 (Event Publisher)                            │
│     └─ 接收和发布事件                                        │

│     ├─ 优先级过滤                                           │
│     ├─ 重复过滤                                             │

│                                                             │
│  4. 事件处理器 (Event Handler)                              │

    ▼
事件消费者
```

### 1.2 事件类型分类

| 类别 | 说明 | 示例 | 优先级 |
|------|------|------|--------|
| **物理事件** | 物理交互、移动、碰撞 | 玩家移动、Agent碰撞 | 中 |
| **对话事件** | Agent间的对话 | Agent说话、玩家对话 | 中 |
| **社会事件** | 社会关系变化、交易 | 结婚、买卖、结盟 | 中 |

| **系统事件** | 系统级别事件 | 区域切换、玩家登出 | 高 |
| **世界事件** | 影响全世界的重大事件 | 国王驾崩、经济崩溃 | 高 |

## 3. 事件发布器 (Event Publisher)

### 3.1 发布接口

```python
class EventPublisher:
    def __init__(self, message_queue, event_store):
        self.queue = message_queue
        self.store = event_store
        self.id_generator = EventIdGenerator()

    def publish(self, event: Event) -> EventId:
        """发布事件"""
        # 1. 验证事件
        self._validate(event)

        # 2. 生成事件ID
        event.id = self.id_generator.generate()

        # 3. 设置时间戳
        event.timestamp = datetime.now(timezone.utc)

        # 4. 持久化
        self.store.save(event)

        # 5. 发送到消息队列
        message = self._to_message(event)
        self.queue.publish(
            topic=self._get_topic(event),
            message=message,
            priority=event.priority.value
        )

        # 6. 触发本地监听器
        self._notify_local(event)

        return event.id

    def publish_batch(self, events: List[Event]) -> List[EventId]:
        """批量发布事件"""
        ids = []
        for event in events:
            ids.append(self.publish(event))
        return ids

    def _validate(self, event: Event):
        """验证事件"""
        if event.event_type is None:
            raise EventValidationError("event_type不能为空")

        if event.source is None:
            raise EventValidationError("source不能为空")

        # 验证事件数据
        self._validate_event_data(event)

    def _get_topic(self, event: Event) -> str:
        """获取事件主题"""
        # 按事件类型分类
        type_prefix = event.event_type.type_prefix()

        # 按优先级分类
        priority_suffix = event.priority.value

        return f"event.{type_prefix}.p{priority_suffix}"
```

## 4. 事件路由器 (Event Router)

### 4.1 路由策略

```python
class EventRouter:
    def __init__(self, subscription_manager):
        self.subscriptions = subscription_manager
        self.routing_rules = []

    def route(self, event: Event) -> List[Subscriber]:
        """路由事件到订阅者"""
        subscribers = []

        # 1. 主题路由
        subscribers.extend(self._route_by_topic(event))

        # 2. 条件路由
        subscribers.extend(self._route_by_condition(event))

        # 3. 地理路由
        subscribers.extend(self._route_by_location(event))

        # 4. 去重
        subscribers = self._deduplicate(subscribers)

        return subscribers

    def _route_by_topic(self, event: Event) -> List[Subscriber]:
        """按主题路由"""
        topic = self._get_event_topic(event)
        return self.subscriptions.get_by_topic(topic)

    def _route_by_condition(self, event: Event) -> List[Subscriber]:
        """按条件路由"""
        matching = []

        for subscription in self.subscriptions.get_conditional():
            if self._match_condition(event, subscription.condition):
                matching.append(subscription.subscriber)

        return matching

    def _route_by_location(self, event: Event) -> List[Subscriber]:
        """按地理位置路由"""
        if event.location is None or event.scope != EventScope.Local:
            return []

        matching = []
        radius = event.scope.radius

        # 查询位置范围内的订阅者
        for subscriber in self.subscriptions.get_location_based():
            distance = self._calculate_distance(event.location, subscriber.position)
            if distance <= radius:
                matching.append(subscriber)

        return matching

    def _match_condition(self, event: Event, condition: Condition) -> bool:
        """匹配条件"""
        # 事件类型匹配
        if condition.event_types and event.event_type not in condition.event_types:
            return False

        # 优先级匹配
        if condition.min_priority and event.priority < condition.min_priority:
            return False

        # 事件数据匹配
        if condition.data_filter:
            if not self._match_data_filter(event.data, condition.data_filter):
                return False

        return True
```

## 5. 事件过滤器 (Event Filter)

### 5.1 过滤链

```python
class EventFilterChain:
    def __init__(self):
        self.filters: List[EventFilter] = []

    def add_filter(self, filter: EventFilter):
        """添加过滤器"""
        self.filters.append(filter)

    def process(self, event: Event) -> FilterResult:
        """处理事件"""
        for filter in self.filters:
            result = filter.filter(event)

            if result.action == FilterAction.Drop:
                # 丢弃事件
                return result

            elif result.action == FilterAction.Modify:
                # 修改事件
                event = result.modified_event

        # 通过所有过滤器
        return FilterResult(action=FilterAction.Pass, event=event)

class EventFilter(ABC):
    @abstractmethod
    def filter(self, event: Event) -> FilterResult:
        pass
```

### 5.2 具体过滤器实现

```python
# 优先级过滤器
class PriorityFilter(EventFilter):
    def __init__(self, min_priority: EventPriority):
        self.min_priority = min_priority

    def filter(self, event: Event) -> FilterResult:
        if event.priority < self.min_priority:
            return FilterResult(
                action=FilterAction.Drop,
                reason=f"优先级过低: {event.priority} < {self.min_priority}"
            )
        return FilterResult(action=FilterAction.Pass, event=event)

# 重复过滤器
class DuplicateFilter(EventFilter):
    def __init__(self, cache_size=10000):
        self.cache = LRUCache(maxsize=cache_size)

    def filter(self, event: Event) -> FilterResult:
        # 生成事件指纹
        fingerprint = self._generate_fingerprint(event)

        if fingerprint in self.cache:
            return FilterResult(
                action=FilterAction.Drop,
                reason="重复事件"
            )

        self.cache[fingerprint] = time.time()
        return FilterResult(action=FilterAction.Pass, event=event)

    def _generate_fingerprint(self, event: Event) -> str:
        """生成事件指纹"""
        key_parts = [
            event.event_type,
            event.source,
            str(event.data),
            str(event.location) if event.location else ""
        ]
        key = "|".join(key_parts)
        return hashlib.md5(key.encode()).hexdigest()

# 速率限制过滤器
class RateLimitFilter(EventFilter):
    def __init__(self):
        self.counters: Dict[str, RateLimiter] = defaultdict(lambda: RateLimiter(limit=10, window=60))

    def filter(self, event: Event) -> FilterResult:
        # 按来源限流
        source_key = str(event.source)
        limiter = self.counters[source_key]

        if not limiter.allow():
            return FilterResult(
                action=FilterAction.Drop,
                reason=f"速率限制: {source_key}"
            )

        return FilterResult(action=FilterAction.Pass, event=event)

class RateLimiter:
    def __init__(self, limit: int, window: int):
        self.limit = limit
        self.window = window
        self.requests: deque = deque()

    def allow(self) -> bool:
        now = time.time()

        # 清理过期请求
        while self.requests and self.requests[0] <= now - self.window:
            self.requests.popleft()

        # 检查是否超限
        if len(self.requests) >= self.limit:
            return False

        self.requests.append(now)
        return True

# 过期过滤器
class ExpirationFilter(EventFilter):
    def filter(self, event: Event) -> FilterResult:
        if event.expires_at and datetime.now(timezone.utc) > event.expires_at:
            return FilterResult(
                action=FilterAction.Drop,
                reason="事件已过期"
            )
        return FilterResult(action=FilterAction.Pass, event=event)
```

## 6. 事件处理器 (Event Handler)

### 6.1 订阅管理

```python
class SubscriptionManager:
    def __init__(self):
        self.subscriptions: Dict[SubscriptionId, Subscription] = {}
        self.topic_index: Dict[str, Set[SubscriptionId]] = defaultdict(set)
        self.location_index: SpatialIndex = None

    def subscribe(self, subscription: Subscription) -> SubscriptionId:
        """添加订阅"""
        sub_id = self._generate_id()
        subscription.id = sub_id

        # 保存订阅
        self.subscriptions[sub_id] = subscription

        # 按主题索引
        for topic in subscription.topics:
            self.topic_index[topic].add(sub_id)

        # 地理订阅
        if subscription.location_based:
            self.location_index.insert(
                entity_id=sub_id,
                position=subscription.position,
                data={"radius": subscription.perception_radius}
            )

        return sub_id

    def unsubscribe(self, subscription_id: SubscriptionId):
        """取消订阅"""
        if subscription_id not in self.subscriptions:
            return

        subscription = self.subscriptions[subscription_id]

        # 从主题索引移除
        for topic in subscription.topics:
            self.topic_index[topic].discard(subscription_id)

        # 从地理索引移除
        if subscription.location_based:
            self.location_index.remove(subscription_id)

        # 删除订阅
        del self.subscriptions[subscription_id]

    def get_by_topic(self, topic: str) -> List[Subscriber]:
        """获取主题的所有订阅者"""
        sub_ids = self.topic_index.get(topic, set())
        return [self.subscriptions[sid].subscriber for sid in sub_ids]
```

### 6.2 异步处理

```python
class AsyncEventHandler:
    def __init__(self, worker_pool_size=10):
        self.worker_pool = ThreadPoolExecutor(max_workers=worker_pool_size)
        self.handlers: Dict[EventType, List[Handler]] = defaultdict(list)

    def register(self, event_type: EventType, handler: Handler):
        """注册事件处理器"""
        self.handlers[event_type].append(handler)

    def handle(self, event: Event):
        """异步处理事件"""
        handlers = self.handlers.get(event.event_type, [])

        if not handlers:
            return

        # 并行处理
        futures = []
        for handler in handlers:
            future = self.worker_pool.submit(handler, event)
            futures.append(future)

        # 等待所有处理器完成（可选）
        # for future in as_completed(futures):
        #     try:
        #         future.result()
        #     except Exception as e:
        #         self._handle_error(event, e)

    def handle_sync(self, event: Event) -> List[HandlerResult]:
        """同步处理事件"""
        results = []
        handlers = self.handlers.get(event.event_type, [])

        for handler in handlers:
            try:
                result = handler(event)
                results.append(result)
            except Exception as e:
                self._handle_error(event, e)
                results.append(HandlerResult(success=False, error=str(e)))

        return results

    def _handle_error(self, event: Event, error: Exception):
        """处理错误"""
        # 记录错误日志
        logging.error(f"处理事件 {event.id} 时出错: {error}")

        # 发送到错误处理系统
        error_event = Event(
            event_type=EventType.System(SystemEventType.HandlerError),
            source=EventSource.System,
            data={"event_id": event.id, "error": str(error)},
            priority=EventPriority.Normal
        )
        self.error_publisher.publish(error_event)
```

```python
        # 查找附近的Agent
        speaker = self.runtime.get_agent(speaker_id)
        nearby_agents = self._find_nearby_agents(speaker, radius=50.0)

        # 通知每个附近的Agent
        for other_agent in nearby_agents:
            # 将对话事件添加到Agent的事件队列
            other_agent.runtime.event_queue.append(event)

    def on_emergency_event(self, event: Event):
        """处理紧急事件"""
        # 高优先级，立即中断所有相关Agent
        affected_agents = self._find_affected_agents(event)

        for agent in affected_agents:
            # 清空当前活动
            agent.state.current_activity = None

            # 强制重新决策
            agent.runtime.force_rethink = True

            # 添加紧急事件到记忆
            memory = Memory(
                content=self._format_emergency_event(event),
                timestamp=datetime.now(),
                importance=1.0,
                tags=["emergency", event.event_type]
            )
            self.runtime.memory.add(agent.id, memory)

    def _find_affected_agents(self, event: Event) -> List[Agent]:
        """查找受事件影响的Agent"""
        affected = []

        # 地理范围内的Agent
        if event.location:
            nearby = self.runtime.spatial_index.query_range(
                center=event.location,
                radius=event.scope.radius if hasattr(event.scope, "radius") else 1000.0
            )
            affected.extend(nearby)

        # 目标Agent
        if event.target:
            if event.target.entity_id:
                target_agent = self.runtime.get_agent(event.target.entity_id)
                if target_agent:
                    affected.append(target_agent)

        return affected
```

```python
class EventStore:
    def __init__(self, db_connection, vector_db):
        self.db = db_connection
        self.vector = vector_db

    def save(self, event: Event):
        """保存事件"""
        # 1. 保存到关系数据库
        self.db.execute(
            "INSERT INTO events (id, type, source, timestamp, data) VALUES (?, ?, ?, ?, ?)",
            (event.id, event.event_type, event.source, event.timestamp, json.dumps(event.data))
        )

        # 2. 生成向量表示（用于相似事件检索）
        if self._should_index(event):
            vector = self._generate_event_vector(event)
            self.vector.insert(
                collection="events",
                id=event.id,
                vector=vector,
                payload={"event_type": event.event_type, "timestamp": event.timestamp}
            )

    def get(self, event_id: EventId) -> Optional[Event]:
        """获取事件"""
        row = self.db.fetch_one(
            "SELECT * FROM events WHERE id = ?",
            (event_id,)
        )

        if row:
            return self._from_row(row)
        return None

    def query(self, filters: EventFilters) -> List[Event]:
        """查询事件"""
        query = "SELECT * FROM events WHERE 1=1"
        params = []

        if filters.event_type:
            query += " AND type = ?"
            params.append(filters.event_type)

        if filters.start_time:
            query += " AND timestamp >= ?"
            params.append(filters.start_time)

        if filters.end_time:
            query += " AND timestamp <= ?"
            params.append(filters.end_time)

        if filters.source:
            query += " AND source = ?"
            params.append(filters.source)

        rows = self.db.fetch_all(query, tuple(params))
        return [self._from_row(row) for row in rows]

    def search_similar(self, event: Event, top_k: int = 10) -> List[Event]:
        """搜索相似事件"""
        vector = self._generate_event_vector(event)

        results = self.vector.search(
            collection="events",
            query_vector=vector,
            top_k=top_k
        )

        events = []
        for result in results:
            event = self.get(result.id)
            if event:
                events.append(event)

        return events
```

### 7.2 事件回放

```python
class EventReplay:
    def __init__(self, event_store):
        self.store = event_store
        self.replay_speed = 1.0

    def replay(self, start_time: DateTime, end_time: DateTime, speed: float = 1.0):
        """回放事件"""
        self.replay_speed = speed

        # 查询时间段内的事件
        events = self.store.query(EventFilters(
            start_time=start_time,
            end_time=end_time,
            order_by="timestamp"
        ))

        # 按时间戳回放
        last_timestamp = start_time
        for event in events:
            # 计算延迟
            delay = (event.timestamp - last_timestamp).total_seconds() / self.replay_speed
            if delay > 0:
                time.sleep(delay)

            # 重放事件
            self._replay_event(event)
            last_timestamp = event.timestamp

    def _replay_event(self, event: Event):
        """重放单个事件"""
        # 发布重放事件（带标记）
        replay_event = Event(
            event_type=event.event_type,
            source=event.source,
            target=event.target,
            data=event.data,
            priority=event.priority,
            timestamp=datetime.now(),  # 使用当前时间
            location=event.location,
            scope=event.scope
        )
        replay_event.data["_replay"] = True  # 标记为回放事件
        replay_event.data["_original_timestamp"] = event.timestamp

        self.publisher.publish(replay_event)
```

## 8. 事件总线完整实现

```python
class EventBus:
    def __init__(self, config: EventBusConfig):
        # 消息队列
        self.message_queue = MessageQueue(config.queue_config)

        # 组件
        self.publisher = EventPublisher(self.message_queue, config.event_store)
        self.router = EventRouter(config.subscription_manager)
        self.filter_chain = EventFilterChain()
        self.handler = AsyncEventHandler(config.worker_pool_size)
        self.store = EventStore(config.db_connection, config.vector_db)

        # 订阅管理
        self.subscriptions = SubscriptionManager()

        # 初始化过滤器
        self._init_filters(config.filters)

        # 启动消费者
        self._start_consumers(config.consumer_count)

    def publish(self, event: Event) -> EventId:
        """发布事件"""
        return self.publisher.publish(event)

    def subscribe(self, topics: List[str], handler: Handler) -> SubscriptionId:
        """订阅事件"""
        subscription = Subscription(
            topics=topics,
            handler=handler,
            subscriber=Subscriber(type="handler", handler=handler)
        )
        return self.subscriptions.subscribe(subscription)

    def _start_consumers(self, count: int):
        """启动消费者"""
        for i in range(count):
            consumer = EventConsumer(
                event_bus=self,
                consumer_id=f"consumer_{i}"
            )
            consumer.start()

    def _init_filters(self, filters_config):
        """初始化过滤器"""
        if filters_config.enable_priority_filter:
            self.filter_chain.add_filter(
                PriorityFilter(min_priority=filters_config.min_priority)
            )

        if filters_config.enable_duplicate_filter:
            self.filter_chain.add_filter(
                DuplicateFilter(cache_size=filters_config.duplicate_cache_size)
            )

        if filters_config.enable_rate_limit:
            self.filter_chain.add_filter(
                RateLimitFilter()
            )

class EventConsumer:
    def __init__(self, event_bus: EventBus, consumer_id: str):
        self.event_bus = event_bus
        self.consumer_id = consumer_id
        self.running = False

    def start(self):
        """启动消费者"""
        self.running = True
        threading.Thread(target=self._consume_loop, daemon=True).start()

    def stop(self):
        """停止消费者"""
        self.running = False

    def _consume_loop(self):
        """消费循环"""
        while self.running:
            try:
                # 从消息队列拉取消息
                message = self.event_bus.message_queue.consume(timeout=1.0)

                if message:
                    # 反序列化为事件
                    event = self._deserialize(message)

                    # 应用过滤器
                    result = self.event_bus.filter_chain.process(event)

                    if result.action == FilterAction.Pass:
                        # 路由事件
                        subscribers = self.event_bus.router.route(result.event)

                        # 处理事件
                        for subscriber in subscribers:
                            if subscriber.type == "handler":
                                self.event_bus.handler.handle(result.event)

            except Exception as e:
                logging.error(f"Consumer {self.consumer_id} 错误: {e}")
```
